```python
from typing import Self

from pydantic import BaseModel, ConfigDict, Field, field_validator, model_validator


class MemoryCard(BaseModel):
    """Compact card returned by the list endpoint."""

    id: int
    event_date: date
    text_preview: str
    mood: str | None
    tags: list[str]


class MemoryDetail(BaseModel):
    """Full memory body returned by the detail and PATCH endpoints."""

    id: int
    event_date: date
    event_tz: str
    text: str
    mood: str | None
    tags: list[str]
    created_at: datetime
    updated_at: datetime


class MemoryListResponse(BaseModel):
    items: list[MemoryCard]
    next_cursor: str | None
    has_more: bool


class MemoryPatch(BaseModel):
    """Partial update body. Distinguishes missing keys from explicit nulls
    via `model_fields_set` — service must call `model_dump(exclude_unset=True)`."""

    model_config = ConfigDict(extra="forbid")

    text: str | None = None
    event_date: date | None = None
    tz: str | None = None
    mood: str | None = None
    tags: list[str] | None = None

    @field_validator("text")
    @classmethod
    def _text_not_empty(cls, v: str | None) -> str | None:
        if v is None:
            return v
        stripped = v.strip()
        if not stripped:
            raise ValueError("text must not be empty")
        return stripped

    @field_validator("tags")
    @classmethod
    def _tags_normalised(cls, v: list[str] | None) -> list[str] | None:
        if v is None:
            return v
        seen: list[str] = []
        for tag in v:
            normalised = tag.strip().lower()
            if normalised and normalised not in seen:
                seen.append(normalised)
        return seen

    @model_validator(mode="after")
    def _at_least_one_field_and_event_date_requires_tz(self) -> Self:
        fields_set = self.model_fields_set
        editable = {"text", "event_date", "mood", "tags"}
        if not (editable & fields_set):
            raise ValueError("at least one editable field must be present")
        if "event_date" in fields_set and "tz" not in fields_set:
            raise ValueError("tz is required when event_date is set")
        return self


def make_text_preview(text: str, *, max_chars: int = 200, max_lines: int = 3) -> str:
    """First N chars or first M lines of the memory text, whichever is shorter."""
    lines = text.splitlines()
    by_line = "\n".join(lines[:max_lines])
    candidate = by_line if len(by_line) <= max_chars else text[:max_chars]
    if len(candidate) < len(text):
        candidate = candidate.rstrip() + "…"
    return candidate
```

```python
from __future__ import annotations

import base64
import binascii
import json
from dataclasses import dataclass
from datetime import date


class CursorDecodeError(ValueError):
    """Raised when a cursor string cannot be decoded into a valid Cursor."""


@dataclass(frozen=True)
class Cursor:
    event_date: date
    id: int


def encode_cursor(cursor: Cursor) -> str:
    payload = json.dumps({"ed": cursor.event_date.isoformat(), "id": cursor.id})
    return base64.urlsafe_b64encode(payload.encode()).decode()


def decode_cursor(token: str) -> Cursor:
    try:
        raw = base64.urlsafe_b64decode(token.encode())
    except (binascii.Error, ValueError) as exc:
        raise CursorDecodeError("cursor is not valid base64") from exc
    try:
        data = json.loads(raw)
    except json.JSONDecodeError as exc:
        raise CursorDecodeError("cursor payload is not valid JSON") from exc
    if not isinstance(data, dict) or "ed" not in data or "id" not in data:
        raise CursorDecodeError("cursor payload missing required keys")
    try:
        return Cursor(event_date=date.fromisoformat(data["ed"]), id=int(data["id"]))
    except (ValueError, TypeError) as exc:
        raise CursorDecodeError("cursor payload has malformed values") from exc
```

```python
from __future__ import annotations

# (keep existing imports above)
from collections.abc import Sequence
from datetime import date as _date
from uuid import UUID

from sqlalchemy import select, tuple_
from sqlalchemy.orm import Session

from app.models.memory import Memory


def list_memories(
    db: Session,
    *,
    user_id: UUID,
    date_from: _date | None = None,
    date_to: _date | None = None,
    tag: str | None = None,
    limit: int = 50,
    cursor: Cursor | None = None,
) -> tuple[Sequence[Memory], bool]:
    """Returns (page items, has_more). Page contains up to `limit` memories
    sorted by (event_date DESC, id DESC), newest first."""
    stmt = (
        select(Memory)
        .where(Memory.user_id == user_id, Memory.deleted_at.is_(None))
        .order_by(Memory.event_date.desc(), Memory.id.desc())
        .limit(limit + 1)  # one extra to detect has_more
    )
    if date_from is not None:
        stmt = stmt.where(Memory.event_date >= date_from)
    if date_to is not None:
        stmt = stmt.where(Memory.event_date <= date_to)
    if tag is not None:
        stmt = stmt.where(Memory.tags.op("&&")([tag]))
    if cursor is not None:
        stmt = stmt.where(
            tuple_(Memory.event_date, Memory.id) < (cursor.event_date, cursor.id)
        )
    rows = db.execute(stmt).scalars().all()
    has_more = len(rows) > limit
    return rows[:limit], has_more
```

```python
from datetime import date
from typing import Annotated

import structlog
from fastapi import APIRouter, Depends, HTTPException, Query, status
from sqlalchemy.orm import Session

from app.core.security import current_user
from app.db.session import get_db
from app.models.user import User
from app.schemas.memory import MemoryCard, MemoryListResponse, make_text_preview
from app.services.memories import (
    Cursor,
    CursorDecodeError,
    decode_cursor,
    encode_cursor,
    list_memories,
)

router = APIRouter(prefix="/memories", tags=["memories"])

log = structlog.get_logger(__name__)


def _to_card(memory) -> MemoryCard:  # type: ignore[no-untyped-def]
    return MemoryCard(
        id=memory.id,
        event_date=memory.event_date,
        text_preview=make_text_preview(memory.text),
        mood=memory.mood,
        tags=list(memory.tags),
    )


@router.get("", response_model=MemoryListResponse)
def list_endpoint(
    user: Annotated[User, Depends(current_user)],
    db: Annotated[Session, Depends(get_db)],
    date_from: Annotated[date | None, Query(alias="from")] = None,
    date_to: Annotated[date | None, Query(alias="to")] = None,
    tag: str | None = None,
    limit: Annotated[int, Query(ge=1, le=200)] = 50,
    cursor: str | None = None,
) -> MemoryListResponse:
    decoded: Cursor | None = None
    if cursor is not None:
        try:
            decoded = decode_cursor(cursor)
        except CursorDecodeError as exc:
            raise HTTPException(
                status_code=status.HTTP_400_BAD_REQUEST,
                detail="invalid_cursor",
            ) from exc

    rows, has_more = list_memories(
        db,
        user_id=user.id,
        date_from=date_from,
        date_to=date_to,
        tag=tag,
        limit=limit,
        cursor=decoded,
    )
    items = [_to_card(row) for row in rows]
    next_cursor = (
        encode_cursor(Cursor(event_date=rows[-1].event_date, id=rows[-1].id))
        if has_more and rows
        else None
    )
    log.info(
        "memory.list_ok",
        user_id=str(user.id),
        count=len(items),
        has_filters=bool(date_from or date_to or tag),
        has_cursor=cursor is not None,
    )
    return MemoryListResponse(items=items, next_cursor=next_cursor, has_more=has_more)
```

```python
from datetime import UTC, datetime as _datetime
from typing import Any


def patch_memory(
    db: Session,
    *,
    user_id: UUID,
    memory_id: int,
    changes: dict[str, Any],
) -> Memory | None:
    """Apply partial update; returns the refreshed row, or None if not found.
    `changes` is the dict from `MemoryPatch.model_dump(exclude_unset=True)`."""
    memory = get_memory(db, user_id=user_id, memory_id=memory_id)
    if memory is None:
        return None

    if "text" in changes:
        memory.text = changes["text"]
    if "mood" in changes:
        memory.mood = changes["mood"]
    if "tags" in changes:
        memory.tags = changes["tags"]
    if "event_date" in changes:
        memory.event_date = changes["event_date"]
        memory.event_tz = changes["tz"]  # validator guarantees presence

    memory.updated_at = _datetime.now(UTC)
    db.commit()
    db.refresh(memory)
    return memory
```

## Out-of-band notes for the implementer

- **Pydantic v2 `model_fields_set`** is the public API for "which fields did the caller actually send?" Used by both `MemoryPatch._at_least_one_field_and_event_date_requires_tz` and the route's `body.model_dump(exclude_unset=True)`. If this surprises you, glance at https://docs.pydantic.dev/latest/concepts/models/#fields-set before changing it.
- **SQLAlchemy `tuple_(...) < (...)`** generates `(event_date, id) < (?, ?)` which Postgres evaluates lexicographically. That's exactly the pagination semantics we want for sort `(event_date DESC, id DESC)`. If a future maintainer changes the sort, the cursor codec AND the WHERE clause must change together.
- **`Memory.tags.op("&&")([tag])`** uses Postgres array overlap. SQLAlchemy doesn't ship a built-in operator for `&&` on `ARRAY(Text)`, so `op(...)` is the canonical escape. The GIN index will be used by the planner.
- **Tests share a module-scoped reset fixture** — order matters within each file. If you add a test, place it in a position consistent with the state assumptions of surrounding tests, or split the file (per phase-04 convention).
- **No image-related code or schema in this phase** — if you find yourself reaching for an image table or column, stop and re-read the design doc. Phase 09 owns it.
- **Lifespan startup probe** still requires the dev DB to be reachable when running the API directly (`make dev`). Tests don't trigger lifespan.
